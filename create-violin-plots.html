emp_data <- read_csv("emp_tab.csv")

long_emp = emp_data[rep(1:nrow(emp_data), emp_data$freq),

long_emp$net <- 0
long_emp$net_type <- "E"

comp_data <- long_emp

# Read the data of random networks.
for (i in 1:3) {
  temp_tab <- read_csv(paste0("rand_tab_", i, ".csv"))
  long_temp <- temp_tab[rep(1:nrow(temp_tab), temp_tab$freq), 
                        -grep("freq", names(temp_tab))]
  long_temp$net <- i
  long_temp$net_type <- "R"
  comp_data <- rbind(comp_data, long_temp)
}

Plot the distribution of total experiences and total observations (without considering the level of harm).

# Rainclouds for repeated measures, continued 
p10 <- ggplot(comp_data, aes(x = total_exp, y = total_obs, fill = net_type)) +
  geom_flat_violin(aes(fill = net_type),position = position_nudge(x = .1, y = 0), 
                   adjust = 1.5, trim = FALSE, alpha = .5, colour = NA)+
  geom_point(aes(x = as.numeric(total_exp)-.15, y = total_obs, colour = net_type), 
             position = position_jitter(width = .05), size = 1, shape = 20) +
  guides(colour=FALSE) +
  geom_boxplot(aes(x = total_exp, y = total_obs, fill = net_type), 
               outlier.shape = NA, alpha = .5, width = .1, colour = "black")+
  scale_colour_brewer(palette = "Dark2")+
  scale_fill_brewer(palette = "Dark2", 
                    name = "Network", 
                    labels = c("Empirical", "Random")) +
  ggtitle("Distribution of total experiences and total observations", 
          subtitle="Empirical network vs. 3 Randomised networks") +
  labs(x = "Total # of victimisation experiences", 
       y = "Total # of observations") +
  theme_light()
